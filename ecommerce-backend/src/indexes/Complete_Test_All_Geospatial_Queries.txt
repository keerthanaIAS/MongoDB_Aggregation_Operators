// ==========================================
// 1. DROP INDEX - SEE COLLSCAN
// ==========================================
db.locations.dropIndexes()

// Test $geoWithin WITHOUT index
db.locations.find({
    location: {
        $geoWithin: {
            $geometry: {
                type: "Polygon",
                coordinates: [[
                    [-74.02, 40.70],
                    [-73.98, 40.70],
                    [-73.98, 40.73],
                    [-74.02, 40.73],
                    [-74.02, 40.70]
                ]]
            }
        }
    }
}).explain("executionStats")
// → COLLSCAN, 10000 docs


// ==========================================
// 2. CREATE INDEX
// ==========================================
db.locations.createIndex({ location: "2dsphere" })


// ==========================================
// 3. $near - Find Nearby (sorted by distance)
// ==========================================
db.locations.find({
    location: {
        $near: {
            $geometry: { type: "Point", coordinates: [-74.006, 40.7128] },
            $maxDistance: 1000
        }
    }
}).limit(5).explain("executionStats")
// → GEO_NEAR_2DSPHERE, 96 docs

// Actual results with distance
db.locations.aggregate([
    {
        $geoNear: {
            near: { type: "Point", coordinates: [-74.006, 40.7128] },
            distanceField: "distance",
            maxDistance: 1000,
            spherical: true
        }
    },
    { $limit: 5 },
    { $project: { name: 1, distance: 1, _id: 0 } }
])
// → Returns: { name: "restaurant_12", distance: 45.2 }


// ==========================================
// 4. $geoWithin - Find Inside Area
// ==========================================
db.locations.find({
    location: {
        $geoWithin: {
            $geometry: {
                type: "Polygon",
                coordinates: [[
                    [-74.02, 40.70],
                    [-73.98, 40.70],
                    [-73.98, 40.73],
                    [-74.02, 40.73],
                    [-74.02, 40.70]
                ]]
            }
        }
    }
}).explain("executionStats")
// → IXSCAN, ~25 docs


// ==========================================
// 5. $geoIntersects - Find Crossing Path
// ==========================================
db.locations.find({
    location: {
        $geoIntersects: {
            $geometry: {
                type: "LineString",
                coordinates: [
                    [-74.02, 40.70],
                    [-73.98, 40.73]
                ]
            }
        }
    }
}).explain("executionStats")
// → IXSCAN, docs along the line


// ==========================================
// 6. $geoWithin with $centerSphere (Circle)
// ==========================================
db.locations.find({
    location: {
        $geoWithin: {
            $centerSphere: [[-74.006, 40.7128], 5 / 3963.2]  // 5 miles radius
        }
    }
}).explain("executionStats")
// → IXSCAN


// ==========================================
// 7. Compound with type filter
// ==========================================
db.locations.createIndex({ location: "2dsphere", type: 1 })

db.locations.find({
    type: "restaurant",
    location: {
        $near: {
            $geometry: { type: "Point", coordinates: [-74.006, 40.7128] },
            $maxDistance: 1000
        }
    }
}).explain("executionStats")
// → GEO_NEAR_2DSPHERE + type filter
